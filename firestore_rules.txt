rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper: Check if user is logged in
    function isAuthenticated() {
      return request.auth != null;
    }

    // Helper: Check if user is an admin (by role or hardcoded email)
    function isAdmin() {
      return isAuthenticated() && 
             (get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin' ||
              request.auth.token.email == 'vishnurajan24766@gmail.com');
    }

    // 1. Subjects Collection (The missing link!)
    match /subjects/{subjectId} {
      allow read: if isAuthenticated();
      allow write: if isAdmin();
    }

    // 2. Users Collection
    match /users/{userId} {
      allow read: if isAuthenticated() && (request.auth.uid == userId || isAdmin());
      allow write: if isAuthenticated() && (request.auth.uid == userId || isAdmin());
    }

    // 3. Feedback Collection
    match /feedback/{feedbackId} {
      allow create: if isAuthenticated();
      allow read, delete: if isAdmin();
    }

    // 4. Exam Reviews Collection
    match /exam_reviews/{reviewId} {
      allow create: if isAuthenticated();
      // Allow user to read their own reviews or admin to read all
      allow read: if isAuthenticated() && (reviewId.startsWith(request.auth.uid) || isAdmin());
      allow update, delete: if isAdmin();
    }

    // 5. Materials Collection (Legacy support)
    match /materials/{document=**} {
      allow read: if isAuthenticated();
      allow write: if isAdmin();
    }

    // Default: Only admins can touch other stuff
    match /{document=**} {
      allow read, write: if isAdmin();
    }
  }
}
